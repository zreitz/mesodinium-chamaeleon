# Grab p values
pval_list <- lapply(de_list, function(de_results) {
  padj <- sapply(de_results, "[", , "padj")
  padj[is.na(padj)] <- 1
  padj
})

sig_ups <- lapply(c("mc", "kn"), function(i) {
  ifelse(pval_list[[i]] < 0.05 & lfc_list[[i]] > log2(1.2), 1, 0)
})

sig_downs <- lapply(c("mc", "kn"), function(i) {
  ifelse(pval_list[[i]] < 0.05 & lfc_list[[i]] < -log2(1.2), 1, 0)
})
